เชื่อมต่อ Supabase
ตั้งค่าครั้งเดียว — ข้อมูลเรียลไทม์ทุกเครื่อง
ขั้นที่ 1 — สร้าง Project
ไปที่ supabase.com → กด Start your project → Sign in with GitHub (ฟรี)
→ New project → ตั้งชื่อ เลือก Region Southeast Asia (Singapore) → สร้าง (รอ ~2 นาที)
ขั้นที่ 2 — สร้าง Tables
ไปที่ SQL Editor (ไอคอนฐานข้อมูล) → New query → วางโค้ดด้านล่าง → Run
-- วางใน SQL Editor แล้วกด Run CREATE TABLE IF NOT EXISTS students ( id TEXT PRIMARY KEY, name TEXT NOT NULL, room TEXT NOT NULL ); CREATE TABLE IF NOT EXISTS homeworks ( num INTEGER PRIMARY KEY, title TEXT NOT NULL, subject TEXT DEFAULT '', max_score INTEGER DEFAULT 100 ); CREATE TABLE IF NOT EXISTS submissions ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, student_id TEXT NOT NULL, hw_num INTEGER NOT NULL, hw_title TEXT, room TEXT, score NUMERIC, max_score INTEGER DEFAULT 100, submitted_at TIMESTAMPTZ DEFAULT now(), UNIQUE(student_id, hw_num) ); CREATE TABLE IF NOT EXISTS settings ( key TEXT PRIMARY KEY, value JSONB ); -- เปิด Realtime ALTER PUBLICATION supabase_realtime ADD TABLE submissions; ALTER PUBLICATION supabase_realtime ADD TABLE students; ALTER PUBLICATION supabase_realtime ADD TABLE homeworks; -- ปิด RLS (สำหรับใช้ภายในโรงเรียน) ALTER TABLE students DISABLE ROW LEVEL SECURITY; ALTER TABLE homeworks DISABLE ROW LEVEL SECURITY; ALTER TABLE submissions DISABLE ROW LEVEL SECURITY; ALTER TABLE settings DISABLE ROW LEVEL SECURITY;
ขั้นที่ 3 — คัดลอก Keys
ไปที่ Project Settings (ไอคอนเฟือง ซ้ายล่าง) → API
→ คัดลอก Project URL และ anon public key ด้านล่างนี้
