ขั้นที่ 1 — สร้าง Project
ไปที่ supabase.com → กด Start your project → Sign in with GitHub (ฟรี)
→ New project → ตั้งชื่อ เลือก Region Southeast Asia (Singapore) → สร้าง (รอ ~2 นาที)
ขั้นที่ 2 — สร้าง Tables
ไปที่ SQL Editor (ไอคอนฐานข้อมูล) → New query → วางโค้ดด้านล่าง → Run
-- วางใน SQL Editor แล้วกด Run -- ตาราง teachers (ระบบ multi-teacher + Supabase Auth) CREATE TABLE IF NOT EXISTS teachers ( id UUID PRIMARY KEY, -- ใช้ auth.users.id จาก Supabase Auth email TEXT NOT NULL UNIQUE, username TEXT UNIQUE, first_name TEXT NOT NULL DEFAULT '', last_name TEXT NOT NULL DEFAULT '', display_name TEXT NOT NULL, status TEXT DEFAULT 'pending', created_at TIMESTAMPTZ DEFAULT now() ); CREATE TABLE IF NOT EXISTS students ( id TEXT PRIMARY KEY, name TEXT NOT NULL, room TEXT NOT NULL, teacher_id TEXT NOT NULL DEFAULT '' ); CREATE TABLE IF NOT EXISTS homeworks ( id SERIAL PRIMARY KEY, num INTEGER NOT NULL, title TEXT NOT NULL, subject TEXT DEFAULT '', max_score INTEGER DEFAULT 100, teacher_id TEXT NOT NULL DEFAULT '' ); CREATE TABLE IF NOT EXISTS submissions ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, student_id TEXT NOT NULL, hw_num INTEGER NOT NULL, hw_title TEXT, room TEXT, score NUMERIC, max_score INTEGER DEFAULT 100, teacher_id TEXT NOT NULL DEFAULT '', submitted_at TIMESTAMPTZ DEFAULT now() ); CREATE TABLE IF NOT EXISTS settings ( key TEXT PRIMARY KEY, value JSONB ); -- เปิด Realtime ALTER PUBLICATION supabase_realtime ADD TABLE submissions; ALTER PUBLICATION supabase_realtime ADD TABLE students; ALTER PUBLICATION supabase_realtime ADD TABLE homeworks; ALTER PUBLICATION supabase_realtime ADD TABLE teachers; -- ปิด RLS (สำหรับใช้ภายในโรงเรียน) ALTER TABLE students DISABLE ROW LEVEL SECURITY; ALTER TABLE homeworks DISABLE ROW LEVEL SECURITY; ALTER TABLE submissions DISABLE ROW LEVEL SECURITY; ALTER TABLE settings DISABLE ROW LEVEL SECURITY; ALTER TABLE teachers DISABLE ROW LEVEL SECURITY; -- (ทางเลือก) Trigger auto-insert teachers เมื่อ signUp ผ่าน Auth -- หากต้องการให้ teachers table อัปเดตอัตโนมัติจาก Auth ให้รันด้วย: -- CREATE OR REPLACE FUNCTION handle_new_teacher() -- RETURNS TRIGGER AS $$ -- BEGIN -- INSERT INTO public.teachers(id, email, display_name, status) -- VALUES (NEW.id, NEW.email, COALESCE(NEW.raw_user_meta_data->>'display_name',''), 'pending') -- ON CONFLICT (id) DO NOTHING; -- RETURN NEW; -- END; -- $$ LANGUAGE plpgsql SECURITY DEFINER; -- CREATE TRIGGER on_auth_teacher_created AFTER INSERT ON auth.users -- FOR EACH ROW EXECUTE FUNCTION handle_new_teacher();
ขั้นที่ 3 — เปิด Email Auth
ไปที่ Authentication → Providers → เปิด Email
✅ เปิด Enable Email provider
⚙️ ตั้ง Site URL ใน Authentication → URL Configuration ให้ตรงกับ URL ของไฟล์นี้
ขั้นที่ 4 — คัดลอก Keys
ไปที่ Project Settings (ไอคอนเฟือง ซ้ายล่าง) → API
→ คัดลอก Project URL และ anon public key ด้านล่างนี้
